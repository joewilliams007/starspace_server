DROP DATABASE db_starspace;
CREATE DATABASE db_starspace;
ALTER DATABASE db_starspace CHARACTER SET utf8mb4 COLLATE utf8mb4_bin;
USE db_starspace;

CREATE TABLE Users (
user_id INT NOT NULL AUTO_INCREMENT,
username varchar(255),
votes int DEFAULT 0, 
bio varchar(2000) DEFAULT NULL,
link varchar(2000) DEFAULT NULL,
email varchar(255),
password varchar(2000),
follows int DEFAULT 0, 
followers int DEFAULT 0, 
timestamp int,
image BOOLEAN DEFAULT false,
image_path varchar(255) DEFAULT null,
account_created DATE,
PRIMARY KEY ( user_id )
);

CREATE TABLE Posts (
post_id INT NOT NULL AUTO_INCREMENT,
votes int DEFAULT 0,
user_id int,
reports int DEFAULT 0, 
comments int DEFAULT 0,
content text,
item1 text DEFAULT NULL,
item2 text DEFAULT NULL,
item1_votes int DEFAULT 0,
item2_votes int DEFAULT 0,
total_votes int DEFAULT 0,
type varchar(255) DEFAULT "post",
link varchar(255),
timestamp int,
image BOOLEAN,
edited BOOLEAN DEFAULT false,
edited_timestamp int DEFAULT null,
image_path varchar(,255),
post_created DATE,
PRIMARY KEY ( post_id )
);



CREATE TABLE Comments (
comment_id INT NOT NULL AUTO_INCREMENT,
user_id int,
post_id int,
comment varchar(2000),
votes int DEFAULT 0,
reports int DEFAULT 0, 
link varchar(255),
image BOOLEAN,
image_path varchar(255),
timestamp int,
comment_created DATE,
PRIMARY KEY ( comment_id )
);

// Vote can be either -1, 0, 1
CREATE TABLE Vote (
vote_id INT NOT NULL AUTO_INCREMENT,
post_id int,
vote int,
user_id int,
timestamp int,
vote_created DATE,
PRIMARY KEY ( vote_id )
);

CREATE TABLE Vote_comment (
vote_id INT NOT NULL AUTO_INCREMENT,
comment_id int,
vote int,
user_id int,
timestamp int,
vote_created DATE,
PRIMARY KEY ( vote_id )
);

CREATE TABLE Vote_poll (
poll_vote_id INT NOT NULL AUTO_INCREMENT,
post_id int,
item int,
user_id int,
timestamp int,
poll_created DATE,
PRIMARY KEY ( poll_vote_id )
);

CREATE TABLE Notif (
notif_id INT NOT NULL AUTO_INCREMENT,
user_id int,
action_id int,
source_user_id int,
type varchar(2000),
seen BOOLEAN DEFAULT false,
app_opened BOOLEAN DEFAULT false,
timestamp int,
notif_created DATE,
PRIMARY KEY ( notif_id )
);

CREATE TABLE Tags (
tag_id INT NOT NULL AUTO_INCREMENT,
post_id int,
tag varchar(255),
PRIMARY KEY ( tag_id )
);

CREATE TABLE Unique_tags (
unique_tag_id INT NOT NULL AUTO_INCREMENT,
tag varchar(255),
usages INT DEFAULT 0,
PRIMARY KEY ( unique_tag_id )
);

follower_id,
